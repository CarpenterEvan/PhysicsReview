
% Components of |M|^2
\begin{align}
	\begin{split}
		% M_total
		\oo{4}\sum_\text{spins}\mathcal{M}_{\mu\b{\mu}\rightarrow{f\b{f}}}&=
		\oo{4}\sum_\text{spins}
		% A^\mu
		\frac{ieQ_f}{s}\qty(\b{v}_a\gamma^\mu u_b)\qty(\b{u}_1\gamma_\mu v_2)+\ldots
		\\
		&\ldots
		% B^\nu
		+\frac{i}{s-m_Z^2}\qty[\b{v}_a\gamma^\nu\qty(g_L^fP_L-g_R^fP_R)u_b]
		\qty[\b{u}_1\gamma_\nu\qty(g_L^fP_L-g_R^fP_R)v_2]
		\\
		&\ldots
		% \slsh{C}
		+\frac{i}{s-m_Z^2}\qty[\b{v}_a\slsh{p}_Z\qty(g_L^fP_L-g_R^fP_R)u_b]
		\qty[\b{u}_1\slsh{p}_Z\qty(g_L^fP_L-g_R^fP_R)v_2]
	\end{split}
	\\
	\begin{split}
		% M_total^*
		\mathcal{M}_{\mu\b{\mu}\rightarrow{f\b{f}}}^*&=
		% A^\sigma
		\frac{-ieQ_f}{s}\qty(\b{u}_b\gamma^\sigma v_a)\qty(\b{v}_2\gamma_\sigma u_1)+\ldots
		\\
		&\ldots
		% B^\rho
		+\frac{-i}{s-m_Z^2}\qty[\b{u}_b\gamma^\rho\qty(g_L^fP_L-g_R^fP_R)v_a]
		\qty[\b{v}_2\gamma_\rho\qty(g_L^fP_L-g_R^fP_R)u_1]
		\\
		&\ldots
		% \slsh{C}^*
		+\frac{-i}{s-m_Z^2}\qty[\b{u}_b\slsh{p}_Z\qty(g_L^fP_L-g_R^fP_R)v_a]
		\qty[\b{v}_2\slsh{p}_Z\qty(g_L^fP_L-g_R^fP_R)u_1]
	\end{split}
\end{align}
\textbf{Key of the substitutions I use in this problem (with $\mu,\nu$ as dummy indices):}
\begin{align*}
	A^\mu
	&\equiv\frac{ieQ_f}{s}\qty(\b{v}_a\gamma^\mu u_b)\qty(\b{u}_1\gamma_\mu v_2)
	\\
	B^\nu
	&\equiv\frac{i}{s-m_Z^2}\qty[\b{v}_a\gamma^\nu\qty(g_L^fP_L-g_R^fP_R)u_b]
	\qty[\b{u}_1\gamma_\nu\qty(g_L^fP_L-g_R^fP_R)v_2]
	\\
	\slsh{C}
	&\equiv\frac{i}{s-m_Z^2}\qty[\b{v}_a\slsh{p}_Z\qty(g_L^fP_L-g_R^fP_R)u_b]
	\qty[\b{u}_1\slsh{p}_Z\qty(g_L^fP_L-g_R^fP_R)v_2]
	\\
	\slsh{p}_Z
	&\equiv\qty(p_a+p_b)_\mu\gamma^\mu
	\\
	G_f
	&\equiv\qty(g_L^f)^2+\qty(g_R^f)^2
\end{align*}
I know these labels might be slightly unorthodox, but I believe the way I use them is unambiguous. 
\\
Label terms by their indices:
\begin{align}
	\mathcal{M}_{\mu\b{\mu}\rightarrow{f\b{f}}}&=A^\mu+B^\nu+\slsh{C}
	\\
	\mathcal{M}_{\mu\b{\mu}\rightarrow{f\b{f}}}^*&=A^\sigma+B^\rho+\slsh{C}^*
	\\
	\oo{4}\sum_\text{spins}\qty|\hat{\mathcal{M}}_{\mu\b{\mu}\rightarrow{f\b{f}}}|^2
	&=\oo{4}\sum_\text{spins}(A^\mu+B^\nu+\slsh{C})(A^\sigma+B^\rho+\slsh{C}^*)
\end{align}
\begin{equation}
	\oo{4}\sum_\text{spins}\qty|\hat{\mathcal{M}}_{\mu\b{\mu}\rightarrow{f\b{f}}}|^2=
	\oo{4}\sum_\text{spins}(
	  A^\mu A^\sigma + A^\mu B^\rho + A^\mu \slsh{C}^*
	+ B^\nu A^\sigma + B^\nu B^\rho + B^\nu \slsh{C}^*
	+ \slsh{C}A^\sigma+\slsh{C}B^\rho+\slsh{C}\slsh{C}^*)
\end{equation}

			4p_{a,\alpha}p_{b,\beta}\qty(\trexp{\alpha}{\mu}{\beta}{\sigma})\cdot\ldots
			\\
			&\ldots\cdot4k^{1,\kappa}k^{2,\eta}\qty(\trexplow{\kappa}{\mu}{\eta}{\sigma})
		\end{split}
		\\
		\begin{split}
			&=

		\end{split}
		\\

				-(p_a\cdot p_b)\qty(k_1\cdot k_2)
				\\&\qquad

		\end{split}
