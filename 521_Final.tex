	\begin{problem}[Coupling Crossing]
		Take a look at problem 10.6 in the textbook. For the Standard Model and the Minimal Supersymmetric Standard Model compute the energy scale where the $SU(2)$ coupling crosses the $U(1)$ coupling, and where the $SU(2)$ coupling crosses the $SU(3)$ coupling. Thus, you should compute four energy scales in this problem. You should compute these numbers analytically first and then plug in the numbers given in problem 10.6 to give numerical values in terms of GeV.
		\answerline

		\begin{equation}
			\mu\dv{g_i}{\mu}=\oo[B_i]{16\pi^2}g^3_i
			\rightarrow\quad
			\int_{g_i(M_Z)}^{g_i}\frac{\d g_i'}{g_i'^3}
			=\oo[B_i]{16\pi^2}\int_{M_Z}^{\mu}\frac{\d \mu'}{\mu'}
			\rightarrow\quad
			\eval{\frac{-1}{2{g'_i}^2}}_{g_i(M_Z)}^{g}
			=\oo[B_i]{16\pi^2}\eval{\ln\mu'}_{M_Z}^{\mu}
		\end{equation}
		\begin{equation}
			\frac{-1}{2{g_i(\mu)}^2}+\frac{1}{2{g_i(M_Z)}^2}
			=\oo[B_i]{16\pi^2}\ln\qty(\frac{\mu}{M_Z})
			\rightarrow
			\boxed{\frac{{g_i(\mu)}^2}{4\pi}
			=\qty[\frac{4\pi}{{g_i(M_Z)}^2}-\oo[B_i]{2\pi}\ln\qty(\frac{\mu}{M_Z})]^{-1}}
		\end{equation}
		$U(1)_Y, SU(2)_L$ coupling crossing, $g_1(\mu)\propto U(1)_Y$ coupling and $g_2(\mu)= SU(2)_L$ coupling. Solving analytically for the intersection between any two couplings $g_i(\mu)$ and $g_j(\mu)$ where $i\neq j$:
		\begin{align}
			\frac{{g_i(\mu_c)}^2}{4\pi}
			=\qty[\frac{4\pi}{{g_i(M_Z)}^2}-\oo[B_i]{2\pi}\ln\qty(\frac{\mu_c}{M_Z})]^{-1}
			&=\frac{{g_j(\mu_c)}^2}{4\pi}=\qty[\frac{4\pi}{{g_j(M_Z)}^2}-\oo[B_j]{2\pi}\ln\qty(\frac{\mu_c}{M_Z})]^{-1}
			\\
			\qty[\frac{4\pi}{{g_i(M_Z)}^2}-\oo[B_i]{2\pi}\ln\qty(\frac{\mu_c}{M_Z})]^{-1}
			&=\qty[\frac{4\pi}{{g_j(M_Z)}^2}-\oo[B_j]{2\pi}\ln\qty(\frac{\mu_c}{M_Z})]^{-1}
			\\
			\frac{4\pi}{{g_i(M_Z)}^2}-\oo[B_i]{2\pi}\ln\qty(\frac{\mu_c}{M_Z})
			&=\frac{4\pi}{{g_j(M_Z)}^2}-\oo[B_j]{2\pi}\ln\qty(\frac{\mu_c}{M_Z})
			\\
			\frac{4\pi}{{g_i(M_Z)}^2}-\frac{4\pi}{{g_j(M_Z)}^2}
			&=-\oo[B_j]{2\pi}\ln\qty(\frac{\mu_c}{M_Z})+\oo[B_i]{2\pi}\ln\qty(\frac{\mu_c}{M_Z})
			\\
			\oo[8\pi^2]{B_i-B_j}\qty(\oo{{g_i(M_Z)}^2}-\oo{{g_j(M_Z)}^2})
			&=
			\ln\qty(\frac{\mu_c}{M_Z})
		\end{align}
		\begin{align}
			\boxed{\mu_c=M_Z\exp{\oo[8\pi^2]{B_i-B_j}\qty(\oo{{g_i(M_Z)}^2}-\oo{{g_j(M_Z)}^2})}}\label{eqn:critical_mu}
		\end{align}
		For the crossing between $SU(2)_L,\ U(1)_Y$:
		\begin{align}
			\mu_c&=M_Z\exp{\oo[8\pi^2]{B_2-B_1}\qty(\oo{{g_2(M_Z)}^2}-\oo{{g_1(M_Z)}^2})}
		\end{align}
		For finding the crossing between  $SU(2)_L,\ SU(3)_c$, I just need to rewrite \ref{eqn:critical_mu} in terms of $\alpha_3(\mu)$:
		\begin{align}
			\mu_c&=M_Z\exp{\oo[2\pi]{B_3-B_2}\qty(\oo{\alpha_3(M_Z)}-\oo[4\pi]{{g_2(M_Z)}^2})}
		\end{align}
		\newpage
		For the Standard model:
		\begin{description}
			\item[i=1] $g_1(M_Z)=0.461$, $B_1=41/10$
			\item[i=2] $g_2(M_Z)=0.652$, $B_2=-19/6$
			\item[i=3] $\alpha_3(M_Z)=0.1185$, $B_3=-7$
		\end{description}
		\begin{align}

			\text{ coupling in SM at}
			\quad
			\coloredanswer{\mu_{c}\approx 1.158\times10^{13}\text{GeV}}
		\end{align}

	\begin{problem}[Exotic Higgs Decay]
		Let us assume that the Higgs boson can decay into two exotic massless $X$-bosons via a $h-X_\alpha-X_\beta$ Feynman rule of $i(g_{\alpha\beta}k_1 \cdot k_2-k_{1\alpha}k_{2\beta})/\Lambda$, where $\Lambda$ is a mass parameter, and $k_1$ ($k_2$) is the outgoing momentum of the $X_\alpha$ ($X_\beta$) boson. Compute the partial width of $h \rightarrow XX$. Determine the value of $\Lambda$ such that the partial width of the Higgs boson into $X$-bosons is the same as the partial width of the Higgs boson into muons. [Note: there are 5th-force issues with an exotic photon being exactly massless, but for this problem you should treat it as effectively massless.]
		\answerline
		\begin{align}
			\feynmandiagram[baseline=(v1.base), horizontal=v1 to v2]{
			v1 -- [boson, edge label=$h$] v2 -- [boson] c [particle=$X_\alpha$],
			d [particle=$X_\beta$] -- [boson] v2,
			};
			\qquad&\longrightarrow\qquad
			\feynmandiagram[baseline=(v1.base), horizontal=v1 to v2]{
			v1 -- [boson, edge label=$h$] v2 -- [boson, momentum'=$k_1$] c [particle=$\epsilon_\alpha$],
			d [particle=$\epsilon_\beta$] -- [boson, reversed momentum'=$k_2$] v2,
			};
		\end{align}
		\begin{align}
			\mathcal{M}
			&=\frac{i(g_{\alpha\beta}\qty(k_1 \cdot k_2)-k_{1\alpha}k_{2\beta})}{\Lambda}\epsilon_1^\alpha\epsilon_2^\beta
			\\
			\sum_\text{spins}\qty|\mathcal{M}|^2&=\sum_\text{spins}\frac{(g_{\alpha\beta}\qty(k_1 \cdot k_2)-k_{1\alpha}k_{2\beta})}{\Lambda}\epsilon_1^\alpha\epsilon_2^\beta
			\frac{(g_{\mu\nu}\qty(k_1 \cdot k_2)-k_{1\mu}k_{2\nu})}{\Lambda}\epsilon_1^{\mu*}\epsilon_2^{\nu*}
			\\
			&=\oo{\Lambda^2}
			\qty(g_{\alpha\beta}\qty(k_1 \cdot k_2)-k_{1\alpha}k_{2\beta})
			\qty(g_{\mu\nu}\qty(k_1 \cdot k_2)-k_{1\mu}k_{2\nu})
			g^{\alpha\mu}g^{\beta\nu}
			\\
			\begin{split}
				\\
				&=\oo{\Lambda^2}
				\Big(
					g^{\alpha\mu}g^{\beta\nu}g_{\alpha\beta}\qty(k_1 \cdot k_2)g_{\mu\nu}\qty(k_1 \cdot k_2)
					-g^{\alpha\mu}g^{\beta\nu}g_{\alpha\beta}\qty(k_1 \cdot k_2)k_{1\mu}k_{2\nu}
					-\ldots\\&\quad\ldots
					-g^{\alpha\mu}g^{\beta\nu}k_{1\alpha}k_{2\beta}g_{\mu\nu}\qty(k_1 \cdot k_2)
					+g^{\alpha\mu}g^{\beta\nu}k_{1\alpha}k_{2\beta}k_{1\mu}k_{2\nu}
				\Big)
			\end{split}
			\\
			\begin{split}
				\\
				&=\oo{\Lambda^2}
				\Big(
					4\qty(k_1 \cdot k_2)^2
					-\qty(k_1 \cdot k_2)\qty(k_1\cdot k_2)
					-\ldots\\&\quad\ldots
					-\qty(k_1\cdot k_2)\qty(k_1 \cdot k_2)
					+k_{1\alpha}k_{2\beta}k_{1}^{\alpha}k_{2}^{\beta}
				\Big)
			\end{split}
			\\
			\begin{split}
				&=\oo[2\qty(k_1 \cdot k_2)^2+(k_1\cdot k_1)(k_2\cdot k_2)]{\Lambda^2}
			\end{split}
		\end{align}
		The outgoing momenta are equal, and the exotic bosons are massless, so the energy should add up to the higgs rest mass:
		\begin{align}
			k_1^\mu&=\oo{2}[m_h,0,0,m_h]
			&
			k_2^\mu&=\oo{2}[m_h,0,0,-m_h]
			\\
			k_1\cdot k_2&=\oo{2}m_h^2
			&
			k_1\cdot k_1&=k_2\cdot k_2=0
		\end{align}
		Plugging this in to the reduced matrix element:
		\begin{align}
			\sum_\text{spins}\qty|\mathcal{M}|^2
			&=\oo[m_h^4]{2\Lambda^2}
		\end{align}
		Now calculating the differential decay width:
		\begin{align}
			\d\Gamma&=\oo{64\pi^2m_h}\qty|\mathcal{M}|^2
			\d\phi_1\d\cos(\theta_1)
			\\
			&=\oo{64\pi^2m_h}\oo[m_h^4]{2\Lambda^2}
			\d\phi_1\d\cos(\theta_1)
			\\
			&=\oo[m_h^3]{128\pi^2\Lambda^2}
			\d\phi_1\d\cos(\theta_1)
		\end{align}
		Using Martin-Wells (6.25) for the angular integrals (and remembering that the particles are indistinguishable):
		\begin{align}
			\Gamma(h\rightarrow XX)&=\oo[m_h^2]{128\pi^2\Lambda^2}
			2\pi
			\\
			\Gamma(h\rightarrow XX)&=\oo[m_h^2]{64\pi\Lambda^2}
		\end{align}
		What we want is to find the $\Lambda$ that makes the following equation true:
		\begin{align}
			1&=\frac{BR(h\rightarrow XX)}{BR(h\rightarrow \mu^+\mu^-)}
			\\
			1&=\oo[m_h^2]{64\pi\Lambda^2}\oo{BR(h\rightarrow \mu^+\mu^-)}
			\\
			\Lambda&=\oo{\sqrt{BR(h\rightarrow \mu^+\mu^-)}}\sqrt{\oo[m_h^2]{64\pi}}
		\end{align}
		From \href{https://pdg.lbl.gov/2023/tables/contents_tables.html}{PDG}: A Higgs with $m_h=125$GeV has a branching ratio to muons 
		$$BR(h\rightarrow\mu^+\mu^-)=\frac{\Gamma(h\rightarrow\mu^+\mu^-)}{\Gamma}=2.6\times10^{-4}.$$
		This means that the mass parameter $\Lambda$ that would make the two branching ratios equal is: 
		\begin{align}
			\Lambda=\sqrt{\frac{(125)^2}{64*\pi*2.6\times10^{-4}}}
			\\
			\coloredanswer{\Lambda=546.71}
		\end{align}
	\end{problem}
