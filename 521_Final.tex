		\end{split}
	\end{align}
	Terms with just one $\epsilon^{\mu\nu\sigma\rho}$ end up summing to zero, so only the diagonal terms survive multiplication:
	\begin{align}
		\begin{split}
			&=\frac{eQ_f}{4s\qty(s-M_Z^2)}
				\Big[
					\qty(g_V^f)^2

						p_a^{\mu}p_b^{\nu}

		\end{split}
		\\
		\begin{split}
			&=\frac{eQ_f}{4s\qty(s-M_Z^2)}
				\Big[
					2\qty(g_V^f)^2
					\qty(
						\qty(p_a\cdot k_1) \qty(p_b\cdot k_2)

					)
					\ldots\\&\qquad\qquad\qquad\ldots
					-16\qty(g_A^f)^2p_{a,\alpha}p_{b,\beta}k_1^{\kappa}k_2^{\eta}
